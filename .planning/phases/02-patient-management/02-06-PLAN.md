**app/services/visit.py** — VisitService with lifecycle management:

class VisitService:
    """Visit lifecycle management with 5-state model."""

    VALID_TRANSITIONS = {
        "scheduled": ["checked-in", "cancelled"],
        "checked-in": ["in-progress", "cancelled"],
        "in-progress": ["completed"],
        "completed": [],
        "cancelled": [],
    }

    VISIT_REASONS = [
        "consultation", "follow-up", "procedure",
        "emergency", "lab", "vaccination", "other",
    ]

    @staticmethod
    async def create_walkin_visit(db, patient_id, department_id, reason, reason_notes=None, created_by=None) -> Visit:
        - Validate reason is in VISIT_REASONS
        - Create Visit with status='scheduled', reason, patient_id, department_id, created_by
        - db.add, flush
        - Auto check-in: status='checked-in', checked_in_at=now (D-34: walk-ins start checked-in)
        - Audit log with action="walkin_visit_created"
        - flush, return visit

    @staticmethod
    async def create_scheduled_visit(db, patient_id, appointment_id, department_id, doctor_id, reason, scheduled_at, created_by) -> Visit:
        - Create Visit with status='scheduled', scheduled_at
        - Audit log
        - flush, return visit

    @staticmethod
    async def transition_status(db, visit_id, new_status, changed_by, reason=None) -> Visit:
        - Get visit, validate transition against VALID_TRANSITIONS
        - Update status and set appropriate timestamp:
          - 'checked-in': checked_in_at=now
          - 'in-progress': started_at=now
          - 'completed': completed_at=now, calculate duration_minutes from checked_in_at to now (D-36)
          - 'cancelled': cancelled_at=now, cancellation_reason=reason
        - If completed: auto-generate VisitSummary (D-38)
        - Audit log with old_status, new_status
        - flush, return visit

    @staticmethod
    async def _calculate_duration(visit: Visit) -> int | None:
        - If checked_in_at and completed_at: return (completed_at - checked_in_at).total_seconds() / 60
        - Return None if timestamps missing

    @staticmethod
    async def _generate_summary(db, visit: Visit) -> VisitSummary:
        - Create VisitSummary linked to visit.id
        - Pre-populate: doctor_name from user lookup, visit date, reason
        - D-38: Auto-generated summary with configurable template (admin can customize later)
        - Return summary

    @staticmethod
    async def get_visits_for_patient(db, patient_id, page=1, page_size=50) -> tuple[list[Visit], int]:
        - select where patient_id, order by created_at.desc() (chronological)
        - Paginate

    @staticmethod
    async def get_visit(db, visit_id) -> Visit | None:

    @staticmethod
    async def create_referral(db, visit_id, data: VisitReferralCreate, created_by) -> VisitReferral:
        - Create VisitReferral with status='pending'
        - Audit log
        - flush, return referral

Endpoints:

   - await db.commit()

2. POST /patients/{patient_id}/visits (status_code=201, response_model=VisitResponse)
   - require_permission("patient", "update")
   - body: VisitCreate with scheduled_at
   - Call VisitService.create_scheduled_visit(db, patient_id, ...)
   - await db.commit()

3. GET /patients/{patient_id}/visits (response_model=list[VisitResponse])

   - Return VisitService.get_visits_for_patient(db, patient_id, page, page_size)

   - require_permission("patient", "read")
