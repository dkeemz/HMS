**app/services/medical_history.py** — Create MedicalHistoryService following rbac.py @staticmethod/async/flush pattern:

class MedicalHistoryService:
    """Medical history CRUD with append-only immutability (D-11)."""

    # ── Allergy methods ─────────────────────────────────────────────

    @staticmethod
    async def create_allergy(db: AsyncSession, patient_id: uuid.UUID, data: AllergyCreate, created_by: uuid.UUID) -> Allergy:
        - Create Allergy with patient_id, data fields, created_by
        - db.add(allergy), await db.flush()
        - AuditService.log_event(db, user_id=created_by, action="allergy_created", resource="allergy", resource_id=str(allergy.id), patient_id=patient_id, why="Medical history entry")
        - await db.flush()
        - Return allergy

    @staticmethod
    async def get_allergies(db: AsyncSession, patient_id: uuid.UUID) -> list[Allergy]:
        - select(Allergy).where(Allergy.patient_id == patient_id, Allergy.status != 'corrected').order_by(Allergy.created_at.desc())
        - Return list (exclude corrected entries from active view)

    @staticmethod
    async def correct_allergy(db: AsyncSession, allergy_id: uuid.UUID, correction_data: AllergyCreate, corrected_by: uuid.UUID) -> Allergy:
        - Get original allergy, verify it belongs to a patient
        - Mark original: status='corrected'
        - Create NEW Allergy record with correction_data, corrected_by=allergy_id (D-11)
        - Audit with action="allergy_corrected"
        - Return new allergy

    # ── Condition methods ───────────────────────────────────────────

    @staticmethod
    async def create_condition(db, patient_id, data: ConditionCreate, created_by) -> Condition: ...
    @staticmethod
    async def get_conditions(db, patient_id) -> list[Condition]: ...
    @staticmethod
    async def correct_condition(db, condition_id, correction_data, corrected_by) -> Condition: ...

    # ── Surgery methods ─────────────────────────────────────────────

    @staticmethod
    async def create_surgery(db, patient_id, data: SurgeryCreate, created_by) -> Surgery: ...
    @staticmethod
    async def get_surgeries(db, patient_id) -> list[Surgery]: ...
    @staticmethod
    async def correct_surgery(db, surgery_id, correction_data, corrected_by) -> Surgery: ...

    # ── Family History methods ──────────────────────────────────────

    @staticmethod
    async def create_family_history(db, patient_id, data: FamilyHistoryCreate, created_by) -> FamilyHistory: ...
    @staticmethod
    async def get_family_history(db, patient_id) -> list[FamilyHistory]: ...
    @staticmethod
    async def correct_family_history(db, fh_id, correction_data, corrected_by) -> FamilyHistory: ...

    # ── Aggregate ───────────────────────────────────────────────────

    @staticmethod
    async def get_all_history(db, patient_id) -> dict:
        - Return {"allergies": [...], "conditions": [...], "surgeries": [...], "family_history": [...]}
        - Used by profile view to display all history at once

All methods follow the pattern:
- @staticmethod, async, accept db: AsyncSession
- Call await db.flush() not commit()
- AuditService.log_event for every mutation
- Raise ValueError for business logic errors
